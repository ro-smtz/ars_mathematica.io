\usepackage{tikz}

\usepackage{pgfplots}

\def\L{7.5} % cm

% -- Cuerpo del documento
\begin{document}

\begin{tikzpicture}[
	% Propiedades generales del dibujo
	line join = round, line cap = round,
]

	% márgenes del dibujo
	\draw[help lines] (0,0) rectangle (\L,\L);

	% definiciones generales
	\tikzmath{
		\width{plot1} = 5.2 cm; \height{plot1} = 3.1 cm;
		\width{plot2} = \width{plot1}; \height{plot2} = 1.0 cm;
		%
		\xmin = -4.0; \xmax = 4.0;
		%
		\ysep = 6mm;
	}	
	
	\begin{axis}[
		name = plotXY,
		at = {(1.28cm,0.8cm)},
		%
		scale only axis,
		width = \width{main},
		height = \height{XY},
		%
		xlabel = $x$, ylabel = $y$,
		%
		enlarge x limits = false,
		xmin = \xmin, xmax = \xmax,
		ymin = -0.28, ymax = 0.28,
		%
		grid = both,
		grid style = {opacity = 0.7},
		%
		xtick distance = 1,
		%
		xticklabels = \empty,
		yticklabels = \empty,
		%
		xlabel style = {
			anchor = center, 
		},
		ylabel style = {
			anchor = center, 
			rotate = -90,
		},
		%
		tick style = {draw = none}
	]
	
		\addplot[
			only marks, mark = *,
			mark options = {
				scale = 0.45,
				draw opacity = 0.0, fill opacity = 0.75,

		]
			table {dataXY.txt};
			
	\end{axis}
